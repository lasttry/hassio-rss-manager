#include <tunables/global>

profile rssmanager flags=(attach_disconnected,mediate_deleted) {
  # Includes
  # Base access to system resources
  #------------------------------------------------
  #include <abstractions/base>

  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay (presente por padrão nos containers do HA)
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,

  # Acesso a runtime e terminal
  /run/{,**} rwk,
  /dev/tty rw,

  # Suporte opcional ao Bashio
  /usr/lib/bashio/** ix,

  # Ficheiros temporários e diretório de dados do add-on
  /tmp/** rwk,
  /data/** rw,

  # Volume /share (caso necessário)
  /share/** rw,

  # Se tiveres um binário/app separado para o serviço principal:
  /usr/bin/rssmanager_app cx -> rssmanager_app,

  profile rssmanager_app flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    signal (receive) peer=*rssmanager,

    /usr/bin/rssmanager_app r,
    /bin/bash rix,
    /bin/echo ix,
    /etc/passwd r,
    /dev/tty rw,
    /data/** rw,
    /share/** rw,
  }
}
